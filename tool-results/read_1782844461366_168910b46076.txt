    63→    63→    63→function AnimatedCounter({ target, duration = 2000 }: { target: number; duration?: number }) {
    64→    64→    64→  const [count, setCount] = useState(0)
    65→    65→    65→
    66→    66→    66→  useEffect(() => {
    67→    67→    67→    if (target === 0) return
    68→    68→    68→    let startTime: number | null = null
    69→    69→    69→    const animate = (timestamp: number) => {
    70→    70→    70→      if (!startTime) startTime = timestamp
    71→    71→    71→      const progress = Math.min((timestamp - startTime) / duration, 1)
    72→    72→    72→      const eased = 1 - Math.pow(1 - progress, 3)
    73→    73→    73→      setCount(Math.floor(eased * target))
    74→    74→    74→      if (progress < 1) requestAnimationFrame(animate)
    75→    75→    75→    }
    76→    76→    76→    requestAnimationFrame(animate)
    77→    77→    77→  }, [target, duration])
    78→    78→    78→
    79→    79→    79→  return <span>{count.toLocaleString('id-ID')}</span>
    80→    80→    80→}
